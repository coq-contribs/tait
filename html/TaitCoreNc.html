Require Export nc.

Set Implicit Arguments.

Module NormalizationProofNc (R:Requirements).

Import R.

Notation App' := (nc2 App).
Notation app' := (nc2 (@app _)).

Lemma app'_ass : forall (l1 l2 l3: context!) ,
 app' (app' l1 l2) l3 = app' l1 (app' l2 l3).

Definition SN (ρs:context!) (ρ:type) (r:term!) :=
  forall k (σs:context!) ,
    (letnc τs' := app' ρs σs in letnc r':= r in F τs' ρ r' k) ->
  { s:term | letnc τs' := app' ρs σs in letnc r':= r in N τs' ρ r' s }.

Definition SA (ρs:context!) (ρ:type) (r:term!) :=
  forall k (σs:context!) ,
    (letnc τs' := app' ρs σs in letnc r':= r in F τs' ρ r' k) ->
  { s:term | letnc τs' := app' ρs σs in letnc r':= r in A τs' ρ r' s }.

Open Scope type_scope.

Fixpoint SC (ρs:context!) (ρ:type) (r:term!) {struct ρ} : Type :=
  (letnc ρs' := ρs in letnc r' := r in TypJ ρs' r' ρ) **
  match ρ with
  | Iota => SN ρs Iota r
  | ρ-->σ =>
    forall s σs, SC (app' ρs σs) ρ s ->
      SC (app' ρs σs) σ (App' r s)
  end.

Lemma SN_ext_ctx : forall ρ ρs σs r,
 (letnc ρs' := ρs in letnc r' := r in TypJ ρs' r' ρ) ->
 SN ρs ρ r -> SN (app' ρs σs) ρ r.

Lemma SC_TypJ :
 forall ρs ρ r, SC ρs ρ r ->
 (letnc ρs' := ρs in letnc r' := r in TypJ ρs' r' ρ).

Lemma SC_ext_ctx : forall ρ ρs σs r,
 SC ρs ρ r -> SC (app' ρs σs) ρ r.

Lemma one :
 forall ρ ρs r,
 (letnc ρs' := ρs in letnc r' := r in TypJ ρs' r' ρ) ->
 (SC ρs ρ r -> SN ρs ρ r)*(SA ρs ρ r -> SC ρs ρ r).

Definition head (rs:list term) :=
 match rs with
  | nil => [0]
  | r::_ => r
 end.

Definition tail (rs:list term) :=
 match rs with
  | nil => nil
  | _::rs => rs
 end.

Notation head' := (nc1 head).
Notation tail' := (nc1 tail).
Notation "'nth'' n" := (nc2 (nth n)) (at level 10).

Fixpoint SCs (σs:context!) (ρs: context) (rs: (list term)!) {struct ρs} : Type :=
 match ρs with
 | nil => letnc rs' := rs in rs' = nil
 | ρ::ρs =>
   (letnc rs' := rs in rs' <> nil) **
   (SC σs ρ (head' rs))*(SCs σs ρs (tail' rs))
end.

Lemma SCs_length : forall σs ρs rs, SCs σs ρs rs ->
  letnc rs' := rs in length ρs = length rs'.

Lemma SCs_nth : forall σs ρs ρ rs r n, n < length ρs ->
  SCs σs ρs rs ->
  SC σs (nth n ρs ρ) (nth' n rs r).

Lemma SCs_ext_ctx : forall σs σs0 ρs rs,
 SCs σs ρs rs -> SCs (app' σs σs0) ρs rs.

Notation "'sub'' r" := (nc1 (sub r)) (at level 10).
Notation support' := (nc1 support).

Lemma three : forall r rs ρs σs ρ,
  SCs σs ρs (support' rs) -> TypJ ρs r ρ ->
  SC σs ρ (sub' r rs).

Lemma SCs_seq : forall ρs (σs:context!) k,
 (letnc sigmas':=σs in length sigmas' = k) ->
  SCs (app' σs (nc ρs)) ρs (nc (map Var (seq k (length ρs)))).

Lemma normalizeTheorem :
 forall ρs ρ r, TypJ ρs r ρ -> { s:term | N ρs ρ r s }.

End NormalizationProofNc.
