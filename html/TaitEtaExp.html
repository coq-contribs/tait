Library TaitEtaExp

Require Export TaitCore.
Require Export Eta.
Require Export WN.

Set Implicit Arguments.

Module RequirementsProofEtaExp <: Requirements.

 Definition abstr (k:nat)(ρ:type)(r:term) :=
  λρ, sub_swap0 r k.

 Inductive N (ρs:context)(ρ:type)(r s:term) : Prop :=
   N_intro : forall t, WN r t -> EtaExp ρs ρ t s -> N ρs ρ r s.

Lemma Ax1 : forall r s ρs ρ σ k, F ρs (ρ-->σ) r k ->
  N (ρs++ext_ctx ρs k ρ) σ (r;k) s ->
  N ρs (ρ-->σ) r (λk:ρ, s).

 Inductive A (ρs:context): type -> term -> term -> Prop :=
 | AVar : forall ρ k, TypJ ρs [k] ρ -> A ρs ρ [k] [k]
 | AApp : forall ρ σ r s r' s',
   A ρs (ρ-->σ) r s ->
   TypJ ρs r' ρ -> N ρs ρ r' s' -> A ρs σ (r;r') (s;s').
  Hint Constructors A.

 Inductive H' : term -> term -> Prop :=
  | H'_intro : forall ρ r s l, H' ((\ρ,r;s);;l) ((sub r s);;l).
  Hint Constructors H'.

 Definition H (ρs:context)(ρ:type)(r s:term) := H' r s.
  Hint Unfold H.

 Inductive Abis (ρs:context) : type -> term -> term -> Prop :=
  | Abis_intros : forall n l l' l'' σs σ,
     WNs l l' -> EtaExps ρs σs l' l'' ->
     TypJ ρs [n] (σs--->σ) ->
     Abis ρs σ ([n];;l) ([n];;l'').

 Lemma A_Abis : forall ρs ρ r s,
  A ρs ρ r s -> Abis ρs ρ r s.

 Lemma Ax2 : forall ρs r s, A ρs Iota r s -> N ρs Iota r s.
 
 Lemma Ax3 : forall ρs ρ k, TypJ ρs [k] ρ -> A ρs ρ [k] [k].

 Lemma Ax4 : forall ρs ρ σ r r' s s',
  TypJ ρs s ρ ->
  A ρs (ρ-->σ) r r' -> N ρs ρ s s' -> A ρs σ (r;s) (r';s').

 Lemma Ax5 :
   forall ρs ρ r s t, H ρs ρ r s -> N ρs ρ s t -> N ρs ρ r t.

 Lemma Ax6 : forall ρs ρ σ r s rs,
  H ρs σ ((sub (λρ,r) rs);s) (sub r ((s::rs)#(rs.

 Lemma Ax7 : forall ρs ρ σ r s t,
  H ρs (ρ-->σ) r s -> H ρs σ (r;t) (s;t).

 Lemma Ax_H_ext_ctx : forall ρs σs ρ r s, TypJ ρs r ρ ->
  H ρs ρ r s -> H (ρs++σs) ρ r s.


End RequirementsProofEtaExp.

Module CompleteEtaExpProof :=
 NormalizationProof(RequirementsProofEtaExp).

 NormalizationProofNc(RequirementsProofEtaExp).
